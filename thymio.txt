var timeflag = 0

callsub wander
sub wander
	call leds.top(0,32,32)
	motor.right.target = 500
	motor.left.target = 500
sub linehold
	call leds.top(0,32,0)
	if  prox.ground.reflected[0] < prox.ground.reflected[1]
	then
		motor.right.target = 500
		motor.left.target = 200
	else
		motor.left.target = 500
		motor.right.target = 200
	end

onevent prox
	if timeflag == 0
		then
			if prox.horizontal[1] < 1000 and prox.horizontal[2] < 1000 and prox.horizontal[3] < 1000 and prox.horizontal[4] < 1000 and prox.horizontal[5]  < 1000
			then
				#Check if on line
				if prox.ground.reflected[0] < 860 or prox.ground.reflected[1] < 860
				then
					#reset/enable wander timer for 5s
					timer.period[1] = 5
					#Follow line
					callsub linehold
				end
		else
			#On object detect
			#Set colour red
			call leds.top(32,0,0)

			#Turn right until curve
			motor.right.target = 500
			motor.left.target = 0
			#Set timer for curve
			timeflag = 1
			timer.period[0] = 300
		end
	end

onevent timer0
	if timeflag == 1 then
		#Disable wander timer
		timer.period[1] = 0
		#Set colour dark blue
		call leds.top(0,0,32)
		#Curve until line found
		motor.left.target = 500
		motor.right.target = 290
		timeflag = 0
	end
	if timeflag == 2 then
		timeflag = 0
	end

onevent timer1
	callsub wander

onevent buttons
	if  button.center == 1
	 then
	 	if  timeflag == 3 then
			timeflag = 0
			callsub wander
		end
		if  timeflag != 3 then
			timeflag = 3
			motor.left.target = 0
			motor.right.target = 0
		end
    end
    button.center = 0
	if  button.backward== 1
	then
		motor.right.target = 500
		motor.left.target = 0
		timeflag = 2
		timer.period[0] = 2000
	end
